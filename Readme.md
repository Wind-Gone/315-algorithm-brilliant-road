  new hashset;
  while(R < str.length)
  	if(R not in hashset)
  		hashset.add(R);
  		R右移；
  	else
  		更新maxsize；
  		hashset.remove(L);

  更新maxsize  //防止最大长度子串的R在最后一个字符
